CREATE TABLE Pokemon (
	Id SERIAL PRIMARY KEY,
	Name VARCHAR(50) NOT NULL,
	BirthDate TIMESTAMP NOT NULL
);

CREATE TABLE Category (
	Id SERIAL PRIMARY KEY,
	Name VARCHAR(50) NOT NULL
);

CREATE TABLE Owner (
	Id SERIAL PRIMARY KEY,
	FirstName VARCHAR(50) NOT NULL,
	LastName VARCHAR(50) NOT NULL,
	Gym VARCHAR(255) NOT NULL
);

CREATE TABLE Reviewer (
	Id SERIAL PRIMARY KEY,
	FirstName VARCHAR(50) NOT NULL,
	LastName VARCHAR(50) NOT NULL
);

CREATE TABLE Country (
	Id SERIAL PRIMARY KEY,
	Name VARCHAR(50) NOT NULL
);

CREATE TABLE Review (
	Id SERIAL PRIMARY KEY,
	Title VARCHAR(100) NOT NULL,
	Text VARCHAR(255),
	Rating INT NOT NULL
);

CREATE TABLE PokemonOwner (
	PokemonId INT REFERENCES Pokemon(Id),
	OwnerId INT REFERENCES Owner(Id),
	PRIMARY KEY (PokemonId, OwnerId)
);

CREATE TABLE PokemonCategory (
	PokemonId INT REFERENCES Pokemon(Id),
	CategoryId INT REFERENCES Category(Id),
	PRIMARY KEY (PokemonId, CategoryId)
);
